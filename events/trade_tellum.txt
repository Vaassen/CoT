namespace = trade_tellum

country_event = {
	id = trade_tellum.1
	title = trade_tellum.1.t
	desc = trade_tellum.1.d
	picture = ECONOMY_eventPicture

	is_triggered_only = yes

	option = {
		name = trade_tellum.1.a
		random_province = {
			limit = {
			owned_by = ROOT
			trade_goods = grain
			}
		add_province_modifier = {
		name = "exotic_trading_post.1"
		duration = -1
		}
	}
}
	option = {
		name = trade_tellum.1.b
		add_years_of_income = 1
	}
}

country_event = {
	id = trade_tellum.2
	title = trade_tellum.2.t
	desc = trade_tellum.2.d
	picture = ECONOMY_eventPicture

	is_triggered_only = yes

	option = {
		name = trade_tellum.2.a
		random_province = {
				limit = {
					owned_by = ROOT
					NOT = {
					trade_goods = damestear
					trade_goods = ivory
					trade_goods = mithril
					trade_goods = copper
					trade_goods = paper
					trade_goods = gems
					}
					OR = {
					has_terrain = mountain
					has_terrain = hills
					has_terrain = cavern
					}
				}
				change_trade_goods = salt
				add_province_modifier = {
				name = "exotic_trading_post_3"
				duration = -1
				}
			}
		}
	option = {
		name = trade_tellum.2.b
		add_years_of_income = 1
	}
}
