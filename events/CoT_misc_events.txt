country_event = {
	id = cot_misc_events.3
	title = cot_misc_events.3.t
	desc = cot_misc_events.3.d
	picture = BIG_STAR_IN_SKY_eventPicture

	fire_only_once = yes

	trigger = {
		primary_culture = green_orc
		owns_core_province = 831 # South Castonath
		owns_core_province = 832 # Lower Castonath
		owns_core_province = 833 # North Castonath
	}

	immediate = {
		if {
			limit = { NOT = { adm_tech = 3 } }
			add_adm_tech = 1
		}
		if {
			limit = { NOT = { dip_tech = 3 } }
			add_dip_tech = 1
		}
		if {
			limit = { NOT = { mil_tech = 3 } }
			add_mil_tech = 1
		}
	}

	option = {
		name = cot_misc_events.3.a

		custom_tooltip = korgus_event_tooltip_1
		change_innovativeness = 50

		define_ruler = {
			name = "Korgus"
			dynasty = "Dookanson"
			#culture = green_orc
			adm = 3
        	dip = 3
        	mil = 6
			age = 55
			claim = 100
		}

		hidden_effect = {
			add_ruler_personality = mythical_conqueror_personality
			add_ruler_personality = strict_personality
			add_ruler_personality = inspiring_leader_personality
		}
		
		hidden_effect = {
			define_ruler_to_general = {
				fire = 4
				shock = 6
				manuever = 6
				siege = 4
			}
		}

		define_heir = {
			dynasty = "Dookanson"
			#culture = green_orc
			adm = 6
        	dip = 6
        	mil = 6
			age = 18
			claim = 100
		}

		custom_tooltip = korgus_event_tooltip_2

		hidden_effect = {

			add_country_modifier = {
				name = cot_orcs_together_stronk
				duration = -1
			}

			every_known_country = {
				if = {
					limit = {
						primary_culture = green_orc
						capital_scope = {
							OR = {
								region = west_castanor_region
								region = inner_castanor_region
								region = south_castanor_region
								region = east_castanor_region
							}
						}
					}
					
					reverse_add_casus_belli = {
						target = ROOT
						type = cb_vassalize_mission
						months = 600
					}

					add_country_modifier = {
						name = cot_orcs_together_stronk
						duration = -1
					}
				}
			}
		}
	}
}

#Turn ruler to powerful general
country_event = {
	id = cot_misc_events.7
	title = cot_misc_events.7.t
	desc = cot_misc_events.7.d
	picture = BIG_BOOK_eventPicture

	is_triggered_only = yes
	hidden = yes

	option = {
		if = {
			limit = {
				mil = 0
				NOT = { mil = 1 }
			}
			define_ruler_to_general = {
				fire = 1
				shock = 6
				manuever = 0
				siege = 1
			}
		}
		else_if = {
			limit = {
				mil = 1
				NOT = { mil = 2 }
			}
			define_ruler_to_general = {
				fire = 1
				shock = 7
				manuever = 0
				siege = 2
			}
		}
		else_if = {
			limit = {
				mil = 2
				NOT = { mil = 3 }
			}
			define_ruler_to_general = {
				fire = 2
				shock = 7
				manuever = 1
				siege = 3
			}
		}
		else_if = {
			limit = {
				mil = 3
				NOT = { mil = 4 }
			}
			define_ruler_to_general = {
				fire = 2
				shock = 8
				manuever = 1
				siege = 4
			}
		}
		else_if = {
			limit = {
				mil = 4
				NOT = { mil = 5 }
			}
			define_ruler_to_general = {
				fire = 2
				shock = 9
				manuever = 2
				siege = 5
			}
		}
		else_if = {
			limit = {
				mil = 5
				NOT = { mil = 6 }
			}
			define_ruler_to_general = {
				fire = 3
				shock = 10
				manuever = 3
				siege = 6
			}
		}
		else_if = {
			limit = {
				mil = 6
			}
			define_ruler_to_general = {
				fire = 3
				shock = 11
				manuever = 4
				siege = 7
			}
		}
		else = {
			convert_ruler_to_general = yes
		}
	}
}
