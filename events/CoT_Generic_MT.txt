					}

				value = 0
			}
		}
	}

	option = {
		name = cot_generic.11.a
		ROOT = {
			add_country_modifier = {

		}

			}
		}
	}
	option = {
		name = cot_generic.11.e
		ROOT = {
			while = {
				limit = {
					NOT = {

					add_base_tax = 1
				}
				change_variable = {
					which = CoTDevCounterTax
					value = 1
				}
			}
		}
	}
