#RP Shop Setup
country_event = {
	id = CoT_RP_Shop.1
	title = CoT_RP_Shop.1.t
	desc = CoT_RP_Shop.1.d

	hidden = yes
	is_triggered_only = yes

	trigger = {
		NOT = {
			has_global_flag = cot_rp_points_set
		}
	}

	option = {
		name = CoT_RP_Shop.1.a
		set_variable = {
			which = cotRpPoints
			value = 0
		}
		set_variable = {
			which = cotRulerCost
			value = 10
		}
		set_variable = {
			which = cotRulerCostImmortal
			value = 15
		}
		set_variable = {
			which = cotHeirCost
			value = 10
		}
		set_variable = {
			which = cotHeirCostImmortal
			value = 15
		}
		set_global_flag = cot_rp_points_set
	}
}
