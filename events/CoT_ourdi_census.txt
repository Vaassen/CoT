#Census start
country_event = {
    id = CoT_ourdi_census_events.1
    hidden = yes
    
    mean_time_to_happen = { days = 1 }
    
    trigger = {
        has_country_modifier = ourdi_survey_office
		NOT = { has_country_flag = started_censuses }
    }
    
    immediate = {
		set_country_flag = started_censuses
		country_event = {
			id = CoT_ourdi_census_events.2
			days = 30
		}
    }
    
    option = { ai_chance = { factor = 1 } }
}

#Census main event 1
country_event = {
	id = CoT_ourdi_census_events.2
	title = "CoT_ourdi_census_events.2.t"
	desc = "CoT_ourdi_census_events.2.d"
	picture = {
		trigger = {
			has_dlc = "Star and Crescent"
			religion_group = muslim
		}
	   picture = muslimgfx_ECONOMY_eventPicture
	}
	picture = {
	   trigger = { NOT = { religion_group = muslim } }
	   picture = OVEREXTENSION_eventPicture
	}
	
	is_triggered_only = yes

	trigger ={ has_country_modifier = ourdi_survey_office }

	option = { #Carry out exhaustive census with extra state officials
		name = "CoT_ourdi_census_events.2.a"
		add_country_modifier = {
			name = doing_exhaustive_census_mod
			duration = 365
		}
		country_event = {
			id = CoT_ourdi_census_events.3
			days = 365
			tooltip = census_exhaustive_tooltip
		}
	}
	option = { #Carry out regular census with some estimations
		name = "CoT_ourdi_census_events.2.b"
		country_event = {
			id = census_events.4
			days = 365
			tooltip = census_regular_tooltip
		}
	}
	option = { #Not this time
		name = "CoT_ourdi_census_events.2.c"
		country_event = {
			id = census_events.2
			days = 3650
			tooltip = census_rejected_tooltip
		}
	}
}
