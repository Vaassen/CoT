cot_shop_stats_price_increase_ruler = {
	if = {
		limit = {
			has_ruler_flag = cot_shop_stat_increase_short
		}
		multiply_variable = {
			which = cotRulerCost
			value = 2
		}
	}
	if = {
		limit = {
			has_ruler_flag = cot_shop_stat_increase_long
		}
		multiply_variable = {
			which = cotRulerCostImmortal
			value = 2
		}
	}
}
