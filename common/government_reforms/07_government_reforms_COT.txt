	}

					}
				}
			}
		}
	}
	removed_effect = {
		hidden_effect = {
			every_owned_province = {
				limit = { has_province_modifier = wine_lord_wine }
				remove_province_modifier = wine_lord_wine
			}
		}
	}
}
