artifice_invention_research_privilege = {
	icon = privilege_Artf_Research
	max_absolutism = 0
	loyalty = 0.05
	influence = 0 #artificer privileges must never affect influence due to the fact influence affects points
	is_valid = {
		NOT = { has_country_flag = remove_artifice_invention_research }
		OR = {
			NOT = { has_estate_privilege = estate_artificers_organization_state }
			has_estate_privilege = artifice_invention_research_privilege
		}
		NOT = { has_estate_privilege = estate_artificers_organization_feiten_shuimu_cadre }
	}
	can_select = {
		artifice_has_points_for_basic = yes
		is_year = 1445
		NOT = { has_estate_privilege = artifice_invention_directed_research }
		NOT = { has_estate_privilege = artifice_invention_directed_research_feiten_precursor_secrets }
	}
	can_revoke = {
		always = no
	}
	on_granted = {
		artifice_granted_basic = yes
		country_event = {
			id = artifice_events.1
		}
	}
	on_revoked = {
		artifice_remove_basic = yes
		#update_artifice_points = yes #is included in the above effect
		#custom_tooltip = decreases_our_artifice_upkeep_by_10
	}
	penalties = {
		prestige_decay = 0.02
		global_tax_modifier = -0.1
	}
	benefits = {
		#foo
	}
	cooldown_years = 10
	on_cooldown_expires = {
		country_event = {
			id = artifice_events.2
		}
	}
	on_invalid = {
		artifice_remove_basic = yes
		clr_country_flag = remove_artifice_invention_research
	}
	ai_will_do = {
		factor = 100 #ai shouldn't use this one. Just too much work getting them to understand it and making it not crash for linux. Their research will be done via decision? Doesnt look like it -Jay
		modifier = {
			factor = 0
			over_artifice_capacity = yes
		}
	}
}

# Invest in Shuimu stuffs
estate_artificers_feiten_invest_engineering_workshop = {
	icon = privilege_grant_autonomy
	influence = 0
	loyalty = 0.05
	land_share = 5
	max_absolutism = -5
	is_valid = {
		has_country_flag = feiten_shuimu_engineering_workshop_unlocked
		NOT = { has_country_flag = feiten_able_to_build_shuimu_extension }
	}
	can_select = {
		any_owned_province = {
			superregion = yanshen_superregion
			NOT = { has_province_modifier = feiten_shuimu_engineering_workshop }
			OR = {
				has_building = university
				has_production_building_trigger = yes
			}
		}
	}
	can_revoke = {
	}
	on_granted = {
		add_dip_power = -33
		if = {
			limit = { monthly_income = 40 }
			add_treasury = -500
		}
		else = { add_years_of_income = -1 }
		custom_tooltip = "The following will apply after the privilege expires:"
		tooltip = { mission_reward_artifice_points_1 = yes }
		random_owned_province = {
			limit = {
				superregion = yanshen_superregion
				NOT = { has_province_modifier = feiten_shuimu_engineering_workshop }
				has_production_building_trigger = yes
			}
			set_province_flag = shuimu_workshop_build_location
			if = {
				limit = {
					event_target:feiten_or_was_feiten_target = {
						check_variable = {
							which = feiten_number_of_shuimu_engineering_cadre_facilities_investments
							value = 5
						}
					}
				}
				tooltip = {
					add_province_modifier = {
						name = feiten_shuimu_engineering_workshop_no_tech_cost
						duration = -1
					}
				}
			}
			else = {
				tooltip = {
					add_province_modifier = {
						name = feiten_shuimu_engineering_workshop
						duration = -1
					}
				}
			}
		}
	}
	
	on_invalid = {
		hidden_effect = { clr_country_flag = feiten_able_to_build_shuimu_extension }
	}
	
	penalties = { }
	
	benefits = { }
	
	cooldown_years = 5
	on_cooldown_expires = {
		set_country_flag = feiten_able_to_build_shuimu_extension
		random_owned_province = {
			limit = { has_province_flag = shuimu_workshop_build_location }
			clr_province_flag = shuimu_workshop_build_location
			if = {
				limit = {
					event_target:feiten_or_was_feiten_target = {
						check_variable = {
							which = feiten_number_of_shuimu_engineering_cadre_facilities_investments
							value = 5
						}
					}
				}
				add_permanent_province_modifier = {
					name = feiten_shuimu_engineering_workshop_no_tech_cost
					duration = -1
				}
			}
			else = {
				add_permanent_province_modifier = {
					name = feiten_shuimu_engineering_workshop
					duration = -1
				}
			}
		}
		change_variable = {
			which = feiten_number_of_shuimu_engineering_cadre_facilities_investments
			value = 1
		}
	}
	ai_will_do = {
		factor = 1
	}
}

# Promote Starry-Eye scholars
estate_artificers_feiten_starry_eye_promotion = {
	icon = privilege_send_emissary_to_pope
	influence = 0.05
	loyalty = 0.05
	land_share = 0
	max_absolutism = -5
	is_valid = { mission_completed = feiten_shuimu_engineering_cadre }
	can_select = { }
	can_revoke = { }
	on_granted = {
		every_known_country = {
			limit = {
				religious_school = {
					group = raheni
					school = starry_eye_school
				}
			}
			add_opinion = {
				who = ROOT
				modifier = increased_religion_opinion
			}
		}
	}

	on_revoked = {
		every_known_country = {
			limit = {
				religious_school = {
					group = raheni
					school = starry_eye_school
				}
			}
			remove_opinion = {
				who = ROOT
				modifier = increased_religion_opinion
			}
		}
	}
	
	on_invalid = {
		every_known_country = {
			limit = {
				religious_school = {
					group = raheni
					school = starry_eye_school
				}
			}
			remove_opinion = {
				who = ROOT
				modifier = increased_religion_opinion
			}
		}
	}
	
	penalties = {
		global_tax_modifier = -0.1
	}
	
	benefits = {
		institution_spread_from_true_faith = 0.2
		innovativeness_gain = 0.25
	}
	
	on_cooldown_expires = { }
	ai_will_do = {
		factor = 1
	}
}
